Here is a great function I found out about.

let rec advance x f n =
	(* If n = 0, we're done. *)
	if n = 0 then x
	else
		let x =
			if n mod 2 = 0 then x
			else f x
		in
		let f = compose f f in
		let n = n / 2 in
		advance x f n

How about that?
